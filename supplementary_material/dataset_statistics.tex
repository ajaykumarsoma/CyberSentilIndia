\subsection{Annotator Details}

\begin{itemize}
\item \textbf{Total Annotators:} 15
\item \textbf{Background:} Cybersecurity professionals (8), Linguists (7)
\item \textbf{Training Duration:} 40 hours per annotator
\item \textbf{Annotation Protocol:} Dual labeling with senior arbitration
\item \textbf{Quality Control:} Weekly calibration sessions
\end{itemize}

\section{Data Collection Sources}

\begin{itemize}
\item \textbf{Primary Source:} cybercrime.gov.in (National Cybercrime Reporting Portal)
\item \textbf{Secondary Sources:} CERT-In, I4C (Indian Cybercrime Coordination Centre)
\item \textbf{Collection Period:} January 2022 -- December 2024
\item \textbf{Geographic Coverage:} All Indian states and union territories
\item \textbf{Anonymization:} All PII removed or masked
\end{itemize}
